    <script>
        let game = {
            clicks: 0, power: 1, auto: 0, rayCost: 50,
            pokedex: [], sfx: true, theme: 'default', allCaught: false
        };

        const monsters = [
            {id: 1, name: "Bulbasaur", img: "001", chance: 0.2},
            {id: 4, name: "Charmander", img: "004", chance: 0.2},
            {id: 7, name: "Squirtle", img: "007", chance: 0.2},
            {id: 133, name: "Eevee", img: "133", chance: 0.17},
            {id: 150, name: "Mewtwo", img: "150", chance: 0.2},
            {id: 25, name: "Pikachu", img: "025", chance: 0.03} // 3% Chance
        ];

        let eventActive = false;
        let eventMult = 1;

        function updateUI() {
            document.getElementById('click-count').innerText = Math.floor(game.clicks);
            document.getElementById('auto-val').innerText = game.auto;
            document.getElementById('ray-cost').innerText = game.rayCost;
            save();
        }

        function save() { localStorage.setItem('pikaBeauFinal', JSON.stringify(game)); }
        function load() {
            let data = localStorage.getItem('pikaBeauFinal');
            if(data) {
                game = JSON.parse(data);
                changeTheme(game.theme);
                checkPokedexUnlock();
            }
            renderPokedex();
            updateUI();
        }

        function startGame() {
            document.getElementById('main-menu').style.display = 'none';
            document.getElementById('game-window').style.display = 'flex';
            load();
            setInterval(() => { game.clicks += (game.auto / 10) * eventMult; updateUI(); }, 100);
            setInterval(spawnPokemon, 30000);
        }

        function doClick(e) {
            let gain = game.power * eventMult;
            game.clicks += gain;
            if(game.sfx) { let s = document.getElementById('pop-sfx'); s.currentTime=0; s.play(); }
            showFloat(e.clientX, e.clientY, `+${gain}`);
            updateUI();
        }

        function spawnPokemon() {
            let rand = Math.random();
            let cumulative = 0;
            let mon = monsters[0];
            
            for (let m of monsters) {
                cumulative += m.chance;
                if (rand < cumulative) { mon = m; break; }
            }

            if (mon.name === "Pikachu") {
                let r = document.getElementById('rare-sfx'); r.currentTime=0; r.play();
            }

            let img = document.createElement('img');
            img.src = `https://assets.pokemon.com/assets/cms2/img/pokedex/full/${mon.img}.png`;
            img.className = 'spawned-pokemon';
            img.style.left = (Math.random() * 60 + 20) + '%';
            img.style.top = (Math.random() * 60 + 20) + '%';
            
            img.onclick = () => {
                if(!game.pokedex.includes(mon.name)) {
                    game.pokedex.push(mon.name);
                    renderPokedex();
                    checkPokedexUnlock();
                }
                img.remove();
            };

            document.getElementById('tab-clicker').appendChild(img);
            setTimeout(() => { if(img) img.remove(); }, 10000);
        }

        function renderPokedex() {
            let html = '';
            monsters.forEach(m => {
                let caught = game.pokedex.includes(m.name);
                html += `<div class="pika-card ${caught ? 'caught' : ''}">
                    <img src="https://assets.pokemon.com/assets/cms2/img/pokedex/full/${m.img}.png" style="width:100%">
                    <div style="font-weight:bold">${caught ? m.name : '???'}</div>
                    <div style="font-size:0.7rem">${m.name === "Pikachu" ? 'LEGENDARY (3%)' : 'Common'}</div>
                </div>`;
            });
            document.getElementById('pokedex-list').innerHTML = html;
        }

        function checkPokedexUnlock() {
            if (game.pokedex.length === monsters.length) {
                game.allCaught = true;
                let btn = document.getElementById('master-theme-btn');
                btn.classList.remove('locked');
                btn.innerText = "Master Ball Theme";
                btn.style.background = "linear-gradient(45deg, #6b21a8, #db2777)";
                btn.onclick = () => changeTheme('master');
            }
        }

        function changeTheme(t) {
            document.body.className = '';
            document.body.classList.add('theme-' + t);
            game.theme = t;
            save();
        }

        function runEvent(name, mult) {
            if(eventActive) return;
            eventActive = true; eventMult = mult;
            let time = 300;
            document.getElementById('hud-name').innerText = name + " ACTIVE";
            document.getElementById('hud-mult').innerText = mult;
            document.getElementById('event-hud').classList.add('active');
            let timer = setInterval(() => {
                time--;
                let m = Math.floor(time/60), s = time%60;
                document.getElementById('hud-time').innerText = `${m}:${s<10?'0':''}${s}`;
                if(time <= 0) { clearInterval(timer); eventActive = false; eventMult = 1; document.getElementById('event-hud').classList.remove('active'); }
            }, 1000);
        }

        function setTab(t) {
            document.querySelectorAll('.tab, .tab-pane').forEach(el => el.classList.remove('active'));
            document.querySelector(`.tab[onclick="setTab('${t}')"]`).classList.add('active');
            document.getElementById(`tab-${t}`).classList.add('active');
        }

        function buyUpgrade() {
            if(game.clicks >= game.rayCost) { game.clicks -= game.rayCost; game.power *= 2; game.rayCost *= 2; updateUI(); }
        }

        function checkAdmin() {
            if(document.getElementById('admin-pass').value === 'PikachuBeau6') {
                document.getElementById('admin-gate').style.display = 'none';
                document.getElementById('admin-controls').style.display = 'block';
            }
        }

        function modResources(add) {
            let a = parseInt(document.getElementById('adm-amt').value) || 0;
            game.clicks += add ? a : -a; updateUI();
        }

        function showFloat(x, y, txt) {
            let el = document.createElement('div'); el.className = 'floating-text'; el.innerText = txt;
            el.style.left = x + 'px'; el.style.top = y + 'px'; document.body.appendChild(el);
            setTimeout(() => el.remove(), 800);
        }

        function hardReset() { if(confirm("Reset everything?")) { localStorage.clear(); location.reload(); } }
    </script>
